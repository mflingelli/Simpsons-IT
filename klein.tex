\begin{figure}
  \centering
  \begin{tikzpicture}
    \newcommand*\radius{2}
    \newcommand*\Radius{4}
    \begin{axis}[view={60}{30}]
      \addplot3[surf, domain=0:2*pi,z buffer=sort]
         ({(\Radius + \radius * cos(deg(y))) * cos(deg(x))},
         {(\Radius + \radius * cos(deg(y))) * sin(deg(x))},
         {\radius * sin(deg(y))});
    \end{axis}
  \end{tikzpicture}
  \caption{Torus}
\end{figure}

\begin{figure}

         {sin(deg(y)) * (1 + x/2 * cos(deg(y)/2))},
         {x/2 * sin(deg(y)/2)});

  \end{tikzpicture}
  \caption{M\"obiusband}
